class EventEmitter{
  constructor() {
    this._eventpool = {}
  }
  on(event, cb) {
    this._eventpool[event] 
    ? this._eventpool[event].push(cb)
    : this._eventpool[event] = [cb]
  }
  emit(event, ...args) {
    if (this._eventpool[event]) {
      this._eventpool[event].map(cb => {
        cb(...args)
      })
    }
  }
  off(event) {
    delete this._eventpool[event]
  }
  once(event, cb) {
    this.on(event, (...args) => {
      cb(...args)
      this.off(event)
    })
  }
}
