def winMost(oppo, horses, n):
  res = 0
  i, j = 0, 0
  oppo.sort()
  horses.sort()

  while i < n and j < n:
    if oppo[i] < horses[j]:
      res += 1
      j += 1
    j += 1

  return res

function fourSumCount(A, B, C, D){
  let d = {}
  let d2 = {}
  let res = 0

  for (let item of A){
    for (let it of B) {
      d[item + it] = d[item + it] ? d[item + it] + 1 : 1
    }
  }

  for (let item of C){
    for (let it of D) {
      d2[item + it] = d2[item + it] ? d2[item + it] + 1 : 1
    }
  }

  for (let item in d) {
    let other = -item
    if (d2[other]) {
      res += d[item] * d2[other]
    }
  }

  return res
}
