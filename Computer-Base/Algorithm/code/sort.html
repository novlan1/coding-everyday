def selectionSort(A):
  n = len(A)
  
  for i in range(n-1):
    min_index = i
    
    for j in range(i + 1, n):
      if A[min_index] > A[j]:
        min_index = j
        
    A[i], A[min_index] = A[min_index], A[i]
  return A

def quick(A):
  n = len(A)
  if n <= 1: return A
  m = A[0]
  left = quick([i for i in A if i < m])
  right = quick([i for i in A if i > m])
  mm = [i for i in A if i == m]
  return left + mm + right

def heapSort(A):

def shellSort(A):
  n = len(A)
  step = 4
  while step > 0:
    for i in range(step, n):
      p = i
      q = i-step
      while q >= 0:
        if A[q] > A[p]:
          A[p], A[q] = A[q], A[p]
          p = q
          q = p - step
        else:
          break
    step -= 1
  return A

def mergeSort(A):
  n = len(A)
  if n <= 1: return A
  num = n // 2
  
  left = mergeSort(A[:num])
  right = mergeSort(A[num:])
  
  res = []
  i, j = 0, 0
  
  while i < len(left) and j < len(right):
    if left[i] < right[j]:
      res.append(left[i])
      i += 1
    else:
      res.append(right[j])
      j += 1
      
  res += left[i:]
  res += right[j:]
  return res

def countingSort(A):
  mi = min(A)
  ma = max(A)
  res = []
  bu = [0] * (ma - mi + 1)
  
  for i in A:
    bu[i - mi] += 1
    
  for i in range(len(bu)):
    if bu[i]:
      res += [i + mi] * bu[i]
  return res

def sort(A, n, k):
  heapD = heapBuild(A[:k], k)

  for i in range(k, n):
    A[i - k] = heapD[0]
    heapD[0] = A[i]
    heap(heapD, 0, k)

    k -= 1
    heap(heapD, 0, k)

  return A

def shortestSubsequence(A):
  if not A: return 0

  n = len(A)
  left, right = 0, 0
  mi = A[n - 1]
  ma = A[0]

  for i in range(1, n):
    if A[i] >= ma:
      ma = A[i]
    else:
      right = i
  
  for j in range(n - 2, -1, -1):
    if A[j] <= mi:
      mi = A[j]
    else:
      left = j
  
  return 0 if (left == 0 and right == 0) else right - left + 1

def maxGap(A):
  n = len(A)
  mi = min(A)
  ma = max(A)
  s = [[] for i in range(n + 1)]
  for i in A:
    s[(i - mi) * n // (ma - mi)].append(i)

  i, j = 0, 1
  res = 0
  while j < len(s):
    while s[j] == []:
      j += 1
    res = max(res, min(s[j]) - max(s[i]))
    i = j
    j += 1

  return res
