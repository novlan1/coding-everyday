  while A:
    q = A.pop(0)

      ope2 = int(s.pop())
      if q == '+':
        res = ope2 + ope1
      elif q == '-':
        res = ope2 - ope1

function lengthOfLongestSubstring(s) {
  let left = 0;
  let lookup = new Set();
  let curLen = 0;
  let maxLen = 0;

  for (let i = 0; i < s.length; i++) {
    curLen += 1;

    while (lookup.has(s[i])) {
      lookup.delete(s[left]);
      left += 1;
      curLen -= 1;
    }
    if (curLen > maxLen) maxLen = curLen;
    lookup.add(s[i]);
  }

  return maxLen;
}

var binaryGap = function(n) {
let last = 30;
  let res = 0;
  let i = 0;
  while (n) {
    if (n & 1) {
      res = Math.max(res, i - last);
      last = i;
    }
    n >>= 1;
    i += 1;
  }
  return res;
};
